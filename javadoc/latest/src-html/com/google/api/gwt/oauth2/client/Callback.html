016    package com.google.api.gwt.oauth2.client;
017    
018    /**
019     * A callback for any asynchronous call that can result in success or failure.
020     * 
021     * @param <T> The type returned on success
022     * @param <F> The type returned on failure
023     * 
024     * @deprecated This class is copied from GWT trunk, and will be included in GWT
025     * 2.4, at which point the usages will transition over to that class instead.
026     */
027    public interface Callback<T, F> {
028    
029      /**
030       * Called when an asynchronous call fails to complete normally.
031       * 
032       * @param reason failure encountered
033       */
034      void onFailure(F reason);
035    
036      /**
037       * Called when an asynchronous call completes successfully.
038       * 
039       * @param result the value returned
040       */
041      void onSuccess(T result);
042    }
